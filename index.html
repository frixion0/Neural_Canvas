Image Generation Example

You can use our proxy endpoint to generate images with various models, including 'flux' and 'uncen'.

async function generateImage() {
  const response = await fetch('https://9000-firebase-studio-1754744124282.cluster-nzwlpk54dvagsxetkvxzbvslyi.cloudworkstations.dev/api/proxy', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      model: 'flux',
      prompt: 'A cute robot waving hello',
    }),
  });

  const data = await response.json();

  if (response.ok) {
    console.log('Image URL:', data.imageUrl);
  } else {
    console.error('Error:', data.error);
  }
}

generateImage();
